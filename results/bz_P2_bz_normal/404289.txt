I want to create an utility for changing generated field and table names, similar to Hibernate's NamingStrategy. I've implemented a SessionCustomizer, similar to what is explained http://dev.eclipse.org/mhonarc/lists/eclipselink-users/msg00094.html.



    public void customize(Session session) {
        for (ClassDescriptor descriptor : session.getDescriptors().values()) {

            for (DatabaseTable table : descriptor.getTables()) {
                rename(table);
            }

            for (DatabaseMapping mapping : descriptor.getMappings()) {
                if (mapping.getFields() != null) {
                    for (DatabaseField field : mapping.getFields()) {
                        rename(field);
                    }
                }
                if (mapping.getField() != null) {
                    rename(mapping.getField());
                }
            }
        }
    }
